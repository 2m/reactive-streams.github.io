190  @Stochastic @Test
191  public void spec103_mustSignalOnMethodsSequentially() throws Throwable {
192    final int iterations = 100;
193    final int elements = 10;
194
195    stochasticTest(iterations, new Function<Integer, Void>() {
196      @Override
197      public Void apply(final Integer runNumber) throws Throwable {
198        activePublisherTest(elements, true, new PublisherTestRun<T>() {
199          @Override
200          public void run(Publisher<T> pub) throws Throwable {
201            final Latch completionLatch = new Latch(env);
202
203            pub.subscribe(new Subscriber<T>() {
204              private Subscription subs;
205              private long gotElements = 0;
206
207              private ConcurrentAccessBarrier concurrentAccessBarrier = new ConcurrentAccessBarrier();
208
209              /**
210               * Concept wise very similar to a {@link org.reactivestreams.tck.TestEnvironment.Latch}, serves to protect
211               * a critical section from concurrent access, with the added benefit of Thread tracking and same-thread-access awareness.
212               *
213               * Since a <i>Synchronous</i> Publisher may choose to synchronously (using the same {@link Thread}) call
214               * {@code onNext} directly from either {@code subscribe} or {@code request} a plain Latch is not enough
215               * to verify concurrent access safety - one needs to track if the caller is not still using the calling thread
216               * to enter subsequent critical sections ("nesting" them effectively).
217               */
218              final class ConcurrentAccessBarrier {
219                private AtomicReference<Thread> currentlySignallingThread = new AtomicReference<Thread>(null);
220                private volatile String previousSignal = null;
221
222                public void enterSignal(String signalName) {
223                  if((!currentlySignallingThread.compareAndSet(null, Thread.currentThread())) && !isSynchronousSignal()) {
224                    env.flop(String.format(
225                      "Illegal concurrent access detected (entering critical section)! " +
226                        "%s emited %s signal, before %s finished its %s signal.",
227                        Thread.currentThread(), signalName, currentlySignallingThread.get(), previousSignal));
228                  }
229                  this.previousSignal = signalName;
230                }
231
232                public void leaveSignal(String signalName) {
233                  currentlySignallingThread.set(null);
234                  this.previousSignal = signalName;
235                }
236
237                private boolean isSynchronousSignal() {
238                  return (previousSignal != null) && Thread.currentThread().equals(currentlySignallingThread.get());
239                }
240
241              }
242
243              @Override
244              public void onSubscribe(Subscription s) {
245                final String signal = "onSubscribe()";
246                concurrentAccessBarrier.enterSignal(signal);
247
248                subs = s;
249                subs.request(1);
250
251                concurrentAccessBarrier.leaveSignal(signal);
252              }
253
254              @Override
255              public void onNext(T ignore) {
256                final String signal = String.format("onNext(%s)", ignore);
257                concurrentAccessBarrier.enterSignal(signal);
258
259                gotElements += 1;
260                if (gotElements <= elements) // requesting one more than we know are in the stream (some Publishers need this)
261                  subs.request(1);
262
263                concurrentAccessBarrier.leaveSignal(signal);
264              }
265
266              @Override
267              public void onError(Throwable t) {
268                final String signal = String.format("onError(%s)", t.getMessage());
269                concurrentAccessBarrier.enterSignal(signal);
270
271                // ignore value
272
273                concurrentAccessBarrier.leaveSignal(signal);
274              }
275
276              @Override
277              public void onComplete() {
278                final String signal = "onComplete()";
279                concurrentAccessBarrier.enterSignal(signal);
280
281                // entering for completeness
282
283                concurrentAccessBarrier.leaveSignal(signal);
284                completionLatch.close();
285              }
286            });
287
288            completionLatch.expectClose(elements * env.defaultTimeoutMillis(), "Expected 10 elements to be drained");
289          }
290        });
291        return null;
292      }
293    });
294  }

582  @Required @Test
583  @Additional(implement = "boundedDepthOfOnNextAndRequestRecursion")
584  public void spec303_mustNotAllowUnboundedRecursion() throws Throwable {
585    final long oneMoreThanBoundedLimit = boundedDepthOfOnNextAndRequestRecursion() + 1;
586
587    activePublisherTest(oneMoreThanBoundedLimit, false, new PublisherTestRun<T>() {
588      @Override
589      public void run(Publisher<T> pub) throws Throwable {
590        final ThreadLocal<Long> stackDepthCounter = new ThreadLocal<Long>() {
591          @Override
592          protected Long initialValue() {
593            return 0L;
594          }
595        };
596
597        final ManualSubscriber<T> sub = new ManualSubscriberWithSubscriptionSupport<T>(env) {
598          @Override
599          public void onNext(T element) {
600            stackDepthCounter.set(stackDepthCounter.get() + 1);
601            super.onNext(element);
602
603            final Long callsUntilNow = stackDepthCounter.get();
604            if (callsUntilNow > boundedDepthOfOnNextAndRequestRecursion()) {
605              env.flop(String.format("Got %d onNext calls within thread: %s, yet expected recursive bound was %d",
606                                     callsUntilNow, Thread.currentThread(), boundedDepthOfOnNextAndRequestRecursion()));
607
608              // stop the recursive call chain
609              return;
610            }
611
612            // request more right away, the Publisher must break the recursion
613            subscription.value().request(1);
614
615            stackDepthCounter.set(stackDepthCounter.get() - 1);
616          }
617        };
618
619        env.subscribe(pub, sub);
620
621        sub.request(1); // kick-off the `request -> onNext -> request -> onNext -> ...`
622
623        sub.nextElementOrEndOfStream();
624        env.verifyNoAsyncErrors();
625      }
626    });
627  }

722  // Verifies rule: https://github.com/reactive-streams/reactive-streams#3.12
723  @Required @Test
724  public void spec312_cancelMustMakeThePublisherToEventuallyStopSignaling() throws Throwable {
725    // the publisher is able to signal more elements than the subscriber will be requesting in total
726    final int publisherElements = 20;
727
728    final int demand1 = 10;
729    final int demand2 = 5;
730    final int totalDemand = demand1 + demand2;
731
732    activePublisherTest(publisherElements, false, new PublisherTestRun<T>() {
733      @Override @SuppressWarnings("ThrowableResultOfMethodCallIgnored")
734      public void run(Publisher<T> pub) throws Throwable {
735        final ManualSubscriber<T> sub = env.newManualSubscriber(pub);
736
737        sub.request(demand1);
738        sub.request(demand2);
739
740        /*
741          NOTE: The order of the nextElement/cancel calls below is very important (!)
742
743          If this ordering was reversed, given an asynchronous publisher,
744          the following scenario would be *legal* and would break this test:
745
746          > AsyncPublisher receives request(10) - it does not emit data right away, it's asynchronous
747          > AsyncPublisher receives request(5) - demand is now 15
748          ! AsyncPublisher didn't emit any onNext yet (!)
749          > AsyncPublisher receives cancel() - handles it right away, by "stopping itself" for example
750          ! cancel was handled hefore the AsyncPublisher ever got the chance to emit data
751          ! the subscriber ends up never receiving even one element - the test is stuck (and fails, even on valid Publisher)
752
753          Which is why we must first expect an element, and then cancel, once the producing is "running".
754         */
755        sub.nextElement();
756        sub.cancel();
757
758        int onNextsSignalled = 1;
759
760        boolean stillBeingSignalled;
761        do {
762          // put asyncError if onNext signal received
763          sub.expectNone();
764          Throwable error = env.dropAsyncError();
765
766          if (error == null) {
767            stillBeingSignalled = false;
768          } else {
769            onNextsSignalled += 1;
770            stillBeingSignalled = true;
771          }
772
773          // if the Publisher tries to emit more elements than was requested (and/or ignores cancellation) this will throw
774          assertTrue(onNextsSignalled <= totalDemand,
775                     String.format("Publisher signalled [%d] elements, which is more than the signalled demand: %d",
776                                   onNextsSignalled, totalDemand));
777
778        } while (stillBeingSignalled);
779      }
780    });
781
782    env.verifyNoAsyncErrors();
783  }

868        ManualSubscriberWithSubscriptionSupport<T> sub = new BlackholeSubscriberWithSubscriptionSupport<T>(env) {

876            if (subscription.isCompleted()) {
